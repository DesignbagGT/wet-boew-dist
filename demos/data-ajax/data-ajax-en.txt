<p>A basic AjaxLoader wrapper for WET-BOEW that inserts AJAXed in content</a>

<section class="wb-prettify all-pre">
	<h2>Insert content after an element (data-ajax-after)</h2>
	<section>
		<h3>Example</h3>
		<section data-ajax-after="ajax/data-ajax-extra.txt" class="original">
			<h4 id="anchor1">Hello World</h4>
			<p>Lorem ipsum dolor sit amet, consectetur adipisicing elit. Cupiditate, dolorem explicabo perferendis vitae magnam.</p>
		</section>
		<section data-ajax-after="ajax/data-ajax-extra-2.txt" class="original">
			<h4 id="anchor1">Hello World - Part 2</h4>
			<p>Lorem ipsum dolor sit amet, consectetur adipisicing elit. Cupiditate, dolorem explicabo perferendis vitae magnam.</p>
		</section>
	</section>
	<section>
		<h3>Coding</h3>
		<pre><code>&lt;section data-ajax-after="ajax/data-ajax-extra.txt" class="original"&gt;
	&lt;h4 id="anchor1"&gt;Hello World&lt;/h4&gt;
	&lt;p&gt;Lorem ipsum dolor sit amet, consectetur adipisicing elit. Cupiditate, dolorem explicabo perferendis vitae magnam.&lt;/p&gt;
&lt;/section&gt;
&lt;section data-ajax-after="ajax/data-ajax-extra-2.txt" class="original"&gt;
	&lt;h4 id="anchor1"&gt;Hello World - Part 2&lt;/h4&gt;
	&lt;p&gt;Lorem ipsum dolor sit amet, consectetur adipisicing elit. Cupiditate, dolorem explicabo perferendis vitae magnam.&lt;/p&gt;
&lt;/section&gt;</code></pre>
</section>
